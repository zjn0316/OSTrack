# 如何为 OSTrack 添加测试/评估数据集接口

本文档详细说明如何在 OSTrack 中添加一个新的测试（评估）数据集接口。

---

## 📋 总体流程

1. **创建数据集类文件** - 在 `lib/test/evaluation/` 下创建新的 `.py` 文件
2. **配置路径** - 在 `lib/test/evaluation/local.py` 中添加数据集路径
3. **添加环境属性** - 在 `lib/test/evaluation/environment.py` 中添加属性定义
4. **注册数据集** - 在 `lib/test/evaluation/datasets.py` 中注册数据集名称
5. **运行评估** - 使用 `tracking/test.py` 进行测试
6. **分析结果** - 使用 `tracking/analysis_results.py` 查看性能指标

---

## 1️⃣ 创建数据集类文件

### 文件位置
```
lib/test/evaluation/your_dataset_name_dataset.py
```

### 必须实现的内容

#### A. 继承 BaseDataset

```python
import numpy as np
from lib.test.evaluation.data import Sequence, BaseDataset, SequenceList
from lib.test.utils.load_text import load_text

class YourDataset(BaseDataset):
    """数据集描述"""
    
    def __init__(self, split='test'):
        """
        参数:
            split - 'train', 'val', 或 'test'（根据数据集实际情况）
        """
        super().__init__()
        self.base_path = self.env_settings.your_dataset_path
        self.split = split
        self.sequence_info_list = self._get_sequence_info_list()
```

#### B. 必须实现的方法

| 方法 | 说明 | 返回值 |
|------|------|--------|
| `get_sequence_list()` | 返回所有序列列表 | `SequenceList` |
| `_construct_sequence(sequence_info)` | 构建单个序列对象 | `Sequence` |
| `_get_sequence_info_list()` | 获取所有序列的信息列表 | `list[dict]` |
| `__len__()` | 返回序列数量 | `int` |

#### C. 序列信息字典格式

```python
sequence_info = {
    "name": "SequenceName",           # 序列名称
    "path": "SequenceName/img",       # 图像相对路径
    "startFrame": 1,                  # 起始帧号
    "endFrame": 100,                  # 结束帧号
    "nz": 4,                          # 帧号位数（如 0001.jpg 则为 4）
    "ext": "jpg",                     # 图像扩展名
    "anno_path": "SequenceName/groundtruth.txt",  # 标注文件路径
    "object_class": "person"          # 目标类别（可选）
}
```

#### D. 完整示例

参考 [`otb100uwbdataset.py`](./otb100uwbdataset.py)：

```python
import numpy as np
from lib.test.evaluation.data import Sequence, BaseDataset, SequenceList
from lib.test.utils.load_text import load_text
import os


class OTB100UWBDataset(BaseDataset):
    """ OTB100_UWB dataset (测试用，仅使用视觉部分)
    
    数据集结构:
    data/OTB100_UWB/
    ├── train/ (57个序列)
    ├── val/ (19个序列)
    └── test/ (19个序列)
        ├── Biker/
        │   ├── 00000001.jpg  (8位数字编号)
        │   └── groundtruth.txt
        ├── ...
        └── list.txt
    """
    def __init__(self, split='test'):
        """
        args:
            split - 'train', 'val', or 'test'
        """
        super().__init__()
        self.base_path = self.env_settings.otb100_uwb_path
        self.split = split
        self.sequence_info_list = self._get_sequence_info_list()

    def get_sequence_list(self):
        return SequenceList([self._construct_sequence(s) for s in self.sequence_info_list])

    def _construct_sequence(self, sequence_info):
        sequence_path = sequence_info['path']
        nz = sequence_info['nz']
        ext = sequence_info['ext']
        start_frame = sequence_info['startFrame']
        end_frame = sequence_info['endFrame']

        init_omit = 0
        if 'initOmit' in sequence_info:
            init_omit = sequence_info['initOmit']

        # 构建图像路径列表
        frames = ['{base_path}/{split}/{sequence_path}/{frame:0{nz}}.{ext}'.format(
            base_path=self.base_path, 
            split=self.split,
            sequence_path=sequence_path, 
            frame=frame_num, 
            nz=nz, 
            ext=ext
        ) for frame_num in range(start_frame+init_omit, end_frame+1)]

        # 构建标注文件路径
        anno_path = '{}/{}/{}'.format(self.base_path, self.split, sequence_info['anno_path'])

        # 读取标注文件（支持多种分隔符）
        ground_truth_rect = load_text(str(anno_path), delimiter=(',', None), dtype=np.float64, backend='numpy')

        return Sequence(sequence_info['name'], frames, 'otb100_uwb', ground_truth_rect[init_omit:,:],
                        object_class=sequence_info.get('object_class', 'unknown'))

    def __len__(self):
        return len(self.sequence_info_list)

    def _get_sequence_info_list(self):
        """从 list.txt 读取序列列表并构建序列信息"""
        split_dir = os.path.join(self.base_path, self.split)
        list_file = os.path.join(split_dir, 'list.txt')
        
        if os.path.exists(list_file):
            # 从 list.txt 读取
            with open(list_file, 'r') as f:
                seq_names = [line.strip() for line in f.readlines() if line.strip()]
        else:
            # 直接扫描目录
            seq_names = [d for d in os.listdir(split_dir) 
                        if os.path.isdir(os.path.join(split_dir, d))]
        
        sequence_info_list = []
        for seq_name in seq_names:
            seq_dir = os.path.join(split_dir, seq_name)
            
            # 检查图像格式和数量
            img_files = [f for f in os.listdir(seq_dir) if f.endswith(('.jpg', '.png', '.jpeg'))]
            if not img_files:
                continue
            
            # 获取第一张和最后一张图片的帧号
            img_numbers = []
            for img in img_files:
                try:
                    num = int(img.split('.')[0])
                    img_numbers.append(num)
                except:
                    pass
            
            if not img_numbers:
                continue
                
            start_frame = min(img_numbers)
            end_frame = max(img_numbers)
            
            # 判断编号位数
            sample_img = img_files[0]
            nz = len(sample_img.split('.')[0])
            ext = sample_img.split('.')[-1]
            
            seq_info = {
                "name": seq_name,
                "path": seq_name,
                "startFrame": start_frame,
                "endFrame": end_frame,
                "nz": nz,
                "ext": ext,
                "anno_path": "{}/groundtruth.txt".format(seq_name),
                "object_class": "unknown"
            }
            sequence_info_list.append(seq_info)
        
        return sequence_info_list
```

## 4️⃣ 注册数据集名称

### 修改文件：`lib/test/evaluation/datasets.py`

在 `dataset_dict` 字典中添加数据集映射：

```python
from collections import namedtuple
import importlib

DatasetInfo = namedtuple('DatasetInfo', ['module', 'class_name', 'kwargs'])

pt = "lib.test.evaluation.%sdataset"  # 模块路径缩写

dataset_dict = dict(
    # ... 其他数据集 ...
    
    your_dataset=DatasetInfo(
        module=pt % "your_dataset_name",  # 模块名（不含 .py）
        class_name="YourDataset",          # 类名
        kwargs=dict(split='test')          # 构造函数参数
    ),
)
```

**示例**:
```python
dataset_dict = dict(
    otb100_uwb=DatasetInfo(
        module=pt % "otb100uwb", 
        class_name="OTB100UWBDataset", 
        kwargs=dict(split='test')
    ),
    otb100_uwb_train=DatasetInfo(
        module=pt % "otb100uwb", 
        class_name="OTB100UWBDataset", 
        kwargs=dict(split='train')
    ),
    otb100_uwb_val=DatasetInfo(
        module=pt % "otb100uwb", 
        class_name="OTB100UWBDataset", 
        kwargs=dict(split='val')
    ),
)
```

## 🔍 常见问题

### Q1: 如何处理不同的图像命名格式？

在 `_construct_sequence` 方法中调整 `nz` 参数：

```python
# 8位数字: 00000001.jpg
nz = 8

# 4位数字: 0001.jpg
nz = 4

# 5位数字: 00001.jpg
nz = 5
```

### Q2: 如何处理不同的标注格式？

使用 `load_text` 函数的不同参数：

```python
# CSV 格式（逗号分隔）
ground_truth_rect = load_text(anno_path, delimiter=',', dtype=np.float64)

# Tab 分隔
ground_truth_rect = load_text(anno_path, delimiter='\t', dtype=np.float64)

# 空格分隔
ground_truth_rect = load_text(anno_path, delimiter=None, dtype=np.float64)
```

### Q3: 如何支持多个 split（train/val/test）？

在 `__init__` 中接收 `split` 参数：

```python
def __init__(self, split='test'):
    self.split = split
    # 根据 split 设置不同的路径
    if split == 'train':
        self.split_path = os.path.join(self.base_path, 'train')
    elif split == 'val':
        self.split_path = os.path.join(self.base_path, 'val')
    elif split == 'test':
        self.split_path = os.path.join(self.base_path, 'test')
```

然后在 `datasets.py` 中注册多个名称：

```python
dataset_dict = dict(
    mydataset_test=DatasetInfo(..., kwargs=dict(split='test')),
    mydataset_val=DatasetInfo(..., kwargs=dict(split='val')),
)
```
